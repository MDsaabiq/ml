# MULTIPLE LINEAR REGRESSION – COMPLETE LAB PROGRAM (ALL STEPS IN ONE)

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score
import statsmodels.api as sm

# -------------------------------------------------
# STEP 1: LOAD DATASET
# -------------------------------------------------
df = pd.read_csv("housing.csv")

print("Top 5 rows:\n", df.head())
print("Bottom 5 rows:\n", df.tail())
print("Shape:\n", df.shape)
print("Data Types:\n", df.dtypes)
print("Summary:\n", df.describe())

# -------------------------------------------------
# STEP 2: PAIRPLOT
# -------------------------------------------------
sns.pairplot(df)
plt.show()

# -------------------------------------------------
# STEP 3: PREPARE X AND Y
# -------------------------------------------------
Y = df.iloc[:, -1]        # Last column as dependent
X = df.iloc[:, :-1]       # All other columns as independent

# -------------------------------------------------
# STEP 4: ENCODE CATEGORICAL DATA
# -------------------------------------------------
X = pd.get_dummies(X, drop_first=True)

print("After Encoding:\n", X.head())

# -------------------------------------------------
# STEP 5: FEATURE SCALING
# -------------------------------------------------
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# -------------------------------------------------
# STEP 6: TRAIN TEST SPLIT
# -------------------------------------------------
X_train, X_test, Y_train, Y_test = train_test_split(
    X_scaled, Y, test_size=0.3, random_state=0
)

# -------------------------------------------------
# STEP 7: TRAIN MODEL
# -------------------------------------------------
model = LinearRegression()
model.fit(X_train, Y_train)

# -------------------------------------------------
# STEP 8: TRAIN PREDICTION & ERROR
# -------------------------------------------------
Y_pred_train = model.predict(X_train)
errors = Y_train - Y_pred_train

# -------------------------------------------------
# STEP 9: RESIDUAL PLOT
# -------------------------------------------------
plt.scatter(Y_pred_train, errors)
plt.xlabel("Predicted Values")
plt.ylabel("Errors")
plt.title("Residual Plot")
plt.show()

# -------------------------------------------------
# STEP 10: TEST PREDICTION
# -------------------------------------------------
Y_pred_test = model.predict(X_test)

# -------------------------------------------------
# STEP 11: MODEL EVALUATION
# -------------------------------------------------
print("MSE:", mean_squared_error(Y_test, Y_pred_test))
print("R2 Score:", r2_score(Y_test, Y_pred_test))

# -------------------------------------------------
# STEP 12: BACKWARD ELIMINATION
# -------------------------------------------------
X_opt = sm.add_constant(X_scaled)
ols_model = sm.OLS(Y, X_opt).fit()

print(ols_model.summary())
